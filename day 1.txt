    int a = 0;

    double d = 0;

//Find area of circle for given radius
#include<stdio.h>
int main (){
    float pi = 3.14;
    float radius = 0;
    printf("Enter a value: \n");
    scanf("%f%f\n",&radius,&pi);
    print("Area of the circle of given radius is = %f\n", pi *radius*radius)
    return 0;
}
