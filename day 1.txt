/* unary increment
under unary increment we ahve 2 categories such as pre incremnet and post increment
pre incremnet: here we are going to increase the value first later we print or operate on the same operand ex: ++a
post increment: here we are going to print or operate first later we increase the value by one. ex: a++ 
example: 

a = 5 
find the value of z.
z = a+++++a---a-a++
z = (a++)+(++a)-(--a)-(a++)
  = 5+7-6-6
  = 0

x = 7
find the value of a
a = x+++--x+++x-x--+x++
a = 21

//Problem: Write a program to check whether an alphabet character entered by the user is a vowel or a consonant.
#include<stdio.h>
int main(){
    char alphabet;
    printf("Enter the alphabet: ");
    scanf("%c",&alphabet);
    switch(alphabet){
        case 'A':
        case 'a': printf("Vowel");
        break;
        case 'E':
        case 'e': printf("Vowel");
        break;
        case 'I':
        case 'i' : printf("Vowel");
        break;
        case 'O':
        case 'o': printf("Vowel");
        break;
        case 'U':
        case 'u': printf("Vowel");
        break;
        default: printf("Consonant");
        return 0;
    }
}

// Write a program to search given key value(21) inside given array where array elements are 3,7,4,21,43,36,11,2.if found key 
//print index value and statement key found.[linear search algorithm]

#include<stdio.h>
int main(){
    int arr[] = {3,7,4,21,43,36,11,2};
    int i;
    int key = 21;
    int size = sizeof(arr)/sizeof(arr[0]);
    for(i=0;i<size;i++){
        if(arr[i]==key){
            printf("Key found at index: %d\n",i);
            break;  
        }   
    }
    return 0;
}

// Write a progam to count number of vowels available in a given string.
// Note: char arr[] = "Hello world"
//       char arr[] = {'H','e','l','l','o','W','o','r','l','d'}
#include<stdio.h>
int main(){
    char arr[]={'d','h','a','t','r','i'};
    int size = sizeof(arr)/sizeof(arr[0]);
    int i;
    int count = 0;
    for(i=0;i<size;i++){
        if(arr[i] == 'a'|| arr[i]=='e'|| arr[i]=='i'||arr[i] =='o'||arr[i] =='u'){
        count+=1;
        }  
    }
    printf("Vowel count: %d\n",count);
    return 0;
}
